			<h2>
				Directions &amp; Parking</h2>

			<p>
				The directions below will lead you to the Highland Street Parking Garage. Parking in the garage costs $2.00 per hour, $10.00 per day maximum, and the first 30 minutes are free. You may also purchase a visitor permit from the Parking &amp; Transportation website at <a href="http://www.unt.edu/transit">www.unt.edu/transit</a> or from the Parking &amp; Transportation Services office located in the Highland Street Parking Garage for $5.00 per day. Metered parking is also available on campus for $1.00 per hour in most locations unless otherwise posted.<br />
				<br />
				<b>Directions from Dallas to Parking Garage</b><br />
				Take I-35E North to Denton. Stay on I-35E once in Denton and exit McCormick Street. Turn right on McCormick. Turn right on Eagle. Turn left on Avenue C and right on Highland Street. The parking garage will be on the right.  <br />
				<br />
				<b>Directions from Ft. Worth to Parking Garage</b><br />
				Take I-35W north to Denton. Exit I-35W onto I-35E (toward Dallas) and exit McCormick Street. Turn left on McCormick. Turn right on Eagle. Turn left onto Avenue C and right on Highland Street. The parking garage will be on the right.  <br />
				<br />
				<b>Directions from DFW to Parking Garage</b><br />
				Take airport tollway north to 121. Exit east on 121 and continue to Lewisville to I-35E. Turn left from 121 to I-35E going North. Continue on I-35E through Lewisville and Lake Dallas. Remain on I-35E in Denton and exit McCormick Street; turn right on McCormick. Turn right on Eagle. Turn left on Avenue C and right on Highland Street. The parking garage will be on the right.  <br />
				<br />
				<b>Directions from Gainesville to Parking </b><b>Garage</b>Take I-35S to Valley View, Sanger and continue to Denton. Continue to the I-35E and I-35W fork; take I-35E to the left. Exit McCormick street, turn left on McCormick. Turn right on Eagle. Turn left on Avenue C and right on Highland Street. The parking garage will be on the right.</p>

var Voter = function() {
 	this.upButton;
    this.downButton;
    this.nid;
    this.inactiveClass = "inactive";
}

Voter.prototype.init = function() {
	this.upButton = document.querySelector(".up-vote");
    this.upButton.addEventListener("click", this.clickHandler.bind(this), false);
    this.downButton = document.querySelector(".down-vote");
    this.downButton.addEventListener("click", this.clickHandler.bind(this), false);
    //Get the node value
    this.nid = document.querySelector(".row.blog").getAttribute("id");//.split("-")[1];
}

Voter.prototype.clickHandler = function(e){
	var url = "/preska2/relationship/add";
    var data = "";
    var element = e.target || e.srcElement;
    //Get Node id
    data += "&nid=" + this.nid;
	//if upButton was pressed, prepare up vote string
    if (element === this.upButton) {
    	this.upButton.classList.remove(this.inactiveClass);
        this.downButton.classList.add(this.inactiveClass);
    	data += "&action=up_vote";
    } else if (element === this.downButton) {
    	this.downButton.classList.remove(this.inactiveClass);
        this.upButton.classList.add(this.inactiveClass);
    	//if downButton was pressed, prepare down vote string
        data += "&action=down_vote";
    }
    
    //Send the Ajax request
    var xmlhttp;
    if (window.XMLHttpRequest)
      {// code for IE7+, Firefox, Chrome, Opera, Safari
      xmlhttp=new XMLHttpRequest();
      }
    else
      {// code for IE6, IE5
      xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");
      }
    xmlhttp.onreadystatechange=function()
      {
      if (xmlhttp.readyState==4 && xmlhttp.status==200)
        {
        document.getElementById("myDiv").innerHTML=xmlhttp.responseText;
        }
      }
    xmlhttp.open("POST", url, true);
    xmlhttp.setRequestHeader("Content-type","application/x-www-form-urlencoded");
    xmlhttp.send(data);
	console.log(data);
}

function startVoter () {
	
    var v = new Voter();
    v.init();

}

window.addEventListener("load", startVoter, false);
