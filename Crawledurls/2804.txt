function Browser() {

  var ua, s, i;

  this.isIE    = false;  // Internet Explorer
  this.isOP    = false;  // Opera
  this.isNS    = false;  // Netscape
  this.version = null;

  ua = navigator.userAgent;

  s = "Opera";
  if ((i = ua.indexOf(s)) >= 0) {
    this.isOP = true;
    this.version = parseFloat(ua.substr(i + s.length));
    return;
  }

  s = "Netscape6/";
  if ((i = ua.indexOf(s)) >= 0) {
    this.isNS = true;
    this.version = parseFloat(ua.substr(i + s.length));
    return;
  }

  // Treat any other "Gecko" browser as Netscape 6.1.

  s = "Gecko";
  if ((i = ua.indexOf(s)) >= 0) {
    this.isNS = true;
    this.version = 6.1;
    return;
  }

  s = "MSIE";
  if ((i = ua.indexOf(s))) {
    this.isIE = true;
    this.version = parseFloat(ua.substr(i + s.length));
    return;
  }
}

var browser = new Browser();

//----------------------------------------------------------------------------

var activeButton = null;

/* [MODIFIED] This code commented out, not needed for activate/deactivate
   on mouseover.

// Capture mouse clicks on the page so any active button can be

if (browser.isIE)

function pageMousedown(event) {

  var el;

  // If there is no active button, exit.

  if (activeButton == null)
    return;

  // Find the element that was clicked on.

  if (browser.isIE)
    el = window.event.srcElement;
  else
    el = (event.target.tagName ? event.target : event.target.parentNode);

  // If the active button was clicked on, exit.

  if (el == activeButton)
    return;

  // If the element is not part of a menu, reset and clear the active

  if (getContainerWith(el, "DIV", "menu") == null) {

function buttonClick(event, menuId) {

  var button;

  // Get the target button element.

  if (browser.isIE)
    button = window.event.srcElement;
  else
    button = event.currentTarget;

  // Blur focus from the link to remove that annoying outline.

  button.blur();

  // Associate the named menu to this button if not already done.
  // Additionally, initialize menu display.

  if (button.menu == null) {
    button.menu = document.getElementById(menuId);
    if (button.menu.isInitialized == null)
      menuInit(button.menu);
	 
  }
  // [MODIFIED] Added for activate/deactivate on mouseover.

  // Set mouseout event handler for the button, if not already done.

  if (button.onmouseout == null)
    button.onmouseout = buttonOrMenuMouseout;

  // Exit if this button is the currently active one.

  if (button == activeButton)
    return false;

  // [END MODIFIED]

  // Reset the currently active button, if any.

  if (activeButton != null)
    resetButton(activeButton);

  // Activate this button, unless it was the currently active one.

  if (button != activeButton) {
    depressButton(button);
    activeButton = button;
  }
  else
    activeButton = null;

  return false;
}

function buttonMouseover(event, menuId) {

  var button;

  // [MODIFIED] Added for activate/deactivate on mouseover.

  // Activates this button's menu if no other is currently active.

  if (activeButton == null) {
    buttonClick(event, menuId);
    return;
  }

  // [END MODIFIED]

  // Find the target button element.

  if (browser.isIE)
    button = window.event.srcElement;
  else
    button = event.currentTarget;

  // If any other button menu is active, make this one active instead.

  if (activeButton != null && activeButton != button)
    buttonClick(event, menuId);


if (browser.isIE) button.menu.style.visibility = "visible";//samer
}

function depressButton(button) {

  var x, y;

  // Update the button's style class to make it look like it's
  // depressed.

  button.className += " menuButtonActive";

  // [MODIFIED] Added for activate/deactivate on mouseover.

  // Set mouseout event handler for the button, if not already done.

  if (button.onmouseout == null)
    button.onmouseout = buttonOrMenuMouseout;
  if (button.menu.onmouseout == null)
    button.menu.onmouseout = buttonOrMenuMouseout;

  // [END MODIFIED]

  // Position the associated drop down menu under the button and
  // show it.

  x = getPageOffsetLeft(button) +150;
  y = getPageOffsetTop(button) + button.offsetHeight -30;

  // For IE, adjust position.

  if (browser.isIE) {
    x += button.offsetParent.clientLeft;
    y += button.offsetParent.clientTop;
  }

  button.menu.style.left = x + "px";
  button.menu.style.top  = y + "px";
  button.menu.style.visibility = "visible";

  // For IE; size, position and show the menu's IFRAME as well.

  if (button.menu.iframeEl != null)
  {
    button.menu.iframeEl.style.left = button.menu.style.left;
    button.menu.iframeEl.style.top  = button.menu.style.top;
    button.menu.iframeEl.style.width  = button.menu.offsetWidth + "px";
    button.menu.iframeEl.style.height = button.menu.offsetHeight + "px";
    button.menu.iframeEl.style.display = "";
  }
}

function resetButton(button) {

  // Restore the button's style class.

  removeClassName(button, "menuButtonActive");

  // Hide the button's menu, first closing any sub menus.

  if (button.menu != null) {
    closeSubMenu(button.menu);
    button.menu.style.visibility = "hidden";

    // For IE, hide menu's IFRAME as well.

    if (button.menu.iframeEl != null)
      button.menu.iframeEl.style.display = "none";
  }
}

  if (browser.isIE)

  else
    menu = event.currentTarget;

  // Close any active sub menu.

  if (menu.activeItem != null)
    closeSubMenu(menu);
}

function menuItemMouseover(event, menuId) {

  var item, menu, x, y;

  // Find the target item element and its parent menu element.

  if (browser.isIE)
    item = getContainerWith(window.event.srcElement, "A", "menuItem");
  else
    item = event.currentTarget;
  menu = getContainerWith(item, "DIV", "menu");

  // Close any active sub menu and mark this one as active.

  if (menu.activeItem != null)
    closeSubMenu(menu);
  menu.activeItem = item;

  // Highlight the item element.

  item.className += " menuItemHighlight";

  // Initialize the sub menu, if not already done.

  if (item.subMenu == null) {
    item.subMenu = document.getElementById(menuId);
    if (item.subMenu.isInitialized == null)
      menuInit(item.subMenu);
  }

  // [MODIFIED] Added for activate/deactivate on mouseover.

  // Set mouseout event handler for the sub menu, if not already done.

  if (item.subMenu.onmouseout == null)
    item.subMenu.onmouseout = buttonOrMenuMouseout;

  // [END MODIFIED]

  // Get position for submenu based on the menu item.

  x = getPageOffsetLeft(item) + item.offsetWidth;
  y = getPageOffsetTop(item);

  // Adjust position to fit in view.

  var maxX, maxY;

  if (browser.isIE) {
    maxX = Math.max(document.documentElement.scrollLeft, document.body.scrollLeft) +
      (document.documentElement.clientWidth != 0 ? document.documentElement.clientWidth : document.body.clientWidth);
    maxY = Math.max(document.documentElement.scrollTop, document.body.scrollTop) +
      (document.documentElement.clientHeight != 0 ? document.documentElement.clientHeight : document.body.clientHeight);
  }
  if (browser.isOP) {
    maxX = document.documentElement.scrollLeft + window.innerWidth;
    maxY = document.documentElement.scrollTop  + window.innerHeight;
  }
  if (browser.isNS) {
    maxX = window.scrollX + window.innerWidth;
    maxY = window.scrollY + window.innerHeight;
  }
  maxX -= item.subMenu.offsetWidth;
  maxY -= item.subMenu.offsetHeight;

  if (x > maxX)
    x = Math.max(0, x - item.offsetWidth - item.subMenu.offsetWidth
      + (menu.offsetWidth - item.offsetWidth));
  y = Math.max(0, Math.min(y, maxY));

  // Position and show the sub menu.

  item.subMenu.style.left       = x + "px";
  item.subMenu.style.top        = y + "px";
  item.subMenu.style.visibility = "visible";

  // For IE; size, position and display the menu's IFRAME as well.

  if (item.subMenu.iframeEl != null)
  {
    item.subMenu.iframeEl.style.left    = item.subMenu.style.left;
    item.subMenu.iframeEl.style.top     = item.subMenu.style.top;
    item.subMenu.iframeEl.style.width   = item.subMenu.offsetWidth + "px";
    item.subMenu.iframeEl.style.height  = item.subMenu.offsetHeight + "px";
    item.subMenu.iframeEl.style.display = "";
  }

  // Stop the event from bubbling.

  if (browser.isIE)
    window.event.cancelBubble = true;
  else
    event.stopPropagation();
}

function closeSubMenu(menu) {

  if (menu == null || menu.activeItem == null)
    return;

  // Recursively close any sub menus.

  if (menu.activeItem.subMenu != null) {
    closeSubMenu(menu.activeItem.subMenu);
    menu.activeItem.subMenu.style.visibility = "hidden";

    // For IE, hide the sub menu's IFRAME as well.

    if (menu.activeItem.subMenu.iframeEl != null)
      menu.activeItem.subMenu.iframeEl.style.display = "none";

    menu.activeItem.subMenu = null;
  }

  // Deactivate the active menu item.

  removeClassName(menu.activeItem, "menuItemHighlight");
  menu.activeItem = null;
}

function buttonOrMenuMouseout(event) {

  var el;

  // If there is no active button, exit.

  if (activeButton == null)
    return;

  // Find the element the mouse is moving to.

  if (browser.isIE)
    el = window.event.toElement;
  else if (event.relatedTarget != null)
      el = (event.relatedTarget.tagName ? event.relatedTarget : event.relatedTarget.parentNode);

  // If the element is not part of a menu, reset the active button.

  if (getContainerWith(el, "DIV", "menu") == null) {
    resetButton(activeButton);
    activeButton = null;
  }
}

//----------------------------------------------------------------------------
// Code to initialize menus.
//----------------------------------------------------------------------------

function menuInit(menu) {
  var itemList, spanList;
  var textEl, arrowEl;
  var itemWidth;
  var w, dw;
  var i, j;

  // For IE, replace arrow characters.

  if (browser.isIE) {
    menu.style.lineHeight = "2.5ex";
    spanList = menu.getElementsByTagName("SPAN");
    for (i = 0; i < spanList.length; i++)
      if (hasClassName(spanList[i], "menuItemArrow")) {
        spanList[i].style.fontFamily = "Webdings";
        spanList[i].firstChild.nodeValue = "4";
      }
  }

  // Find the width of a menu item.

  itemList = menu.getElementsByTagName("A");
  if (itemList.length > 0)
    itemWidth = itemList[0].offsetWidth;
  else
    return;

  // For items with arrows, add padding to item text to make the
  // arrows flush right.

  for (i = 0; i < itemList.length; i++) {
    spanList = itemList[i].getElementsByTagName("SPAN");
    textEl  = null;
    arrowEl = null;
    for (j = 0; j < spanList.length; j++) {
      if (hasClassName(spanList[j], "menuItemText"))
        textEl = spanList[j];
      if (hasClassName(spanList[j], "menuItemArrow"))
        arrowEl = spanList[j];
    }
    if (textEl != null && arrowEl != null) {
      textEl.style.paddingRight = (itemWidth 
        - (textEl.offsetWidth + arrowEl.offsetWidth)) + "px";
      // For Opera, remove the negative right margin to fix a display bug.
      if (browser.isOP)
        arrowEl.style.marginRight = "0px";
    }
  }

  // Fix IE hover problem by setting an explicit width on first item of
  // the menu.

  if (browser.isIE) {
    w = itemList[0].offsetWidth;
    itemList[0].style.width = w + "px";
    dw = itemList[0].offsetWidth - w;
    w -= dw;
    itemList[0].style.width = w + "px";
  }

  // Fix the IE display problem (SELECT elements and other windowed controls
  // overlaying the menu) by adding an IFRAME under the menu.

  if (browser.isIE) {
    menu.iframeEl = menu.parentNode.insertBefore(document.createElement("IFRAME"), menu);
    menu.iframeEl.style.display = "none";
    menu.iframeEl.style.position = "absolute";
	 menu.iframeEl.style.visibility = "visible";
  }

  // Mark menu as initialized.
  menu.isInitialized = true;
}

//----------------------------------------------------------------------------
// General utility functions.
//----------------------------------------------------------------------------

function getContainerWith(node, tagName, className) {

  // Starting with the given node, find the nearest containing element
  // with the specified tag name and style class.

  while (node != null) {
    if (node.tagName != null && node.tagName == tagName &&
        hasClassName(node, className))
      return node;
    node = node.parentNode;
  }

  return node;
}

function hasClassName(el, name) {

  var i, list;

  // Return true if the given element currently has the given class
  // name.

  list = el.className.split(" ");
  for (i = 0; i < list.length; i++)
    if (list[i] == name)
      return true;

  return false;
}

function removeClassName(el, name) {

  var i, curList, newList;

  if (el.className == null)
    return;

  // Remove the given class name from the element's className property.

  newList = new Array();
  curList = el.className.split(" ");
  for (i = 0; i < curList.length; i++)
    if (curList[i] != name)
      newList.push(curList[i]);
  el.className = newList.join(" ");
}

function getPageOffsetLeft(el) {

  var x;
  // Return the x coordinate of an element relative to the page.
  x = el.offsetLeft;
  if (el.offsetParent != null)
    x += getPageOffsetLeft(el.offsetParent);
  return x;
}

function getPageOffsetTop(el) {

  var y;

  // Return the x coordinate of an element relative to the page.

  y = el.offsetTop;
  if (el.offsetParent != null)
    y += getPageOffsetTop(el.offsetParent);

  return y;
}

	<p><strong><a name="AnotherUniversity" class="DefaultLink">I am pursuing a degree at another university, but would like to register for one or more online courses at UNT.  How do I register for courses at UNT?</a></strong></p>
	<p>Transient  students must apply for admission to UNT before registering for courses.&nbsp;  You can apply online at <a href="http://www.applytexas.org/" class="DefaultLink">http://www.applytexas.org</a> or see  the Admissions page, <a href="http://www.unt.edu/vwbk/admission.htm" class="DefaultLink">http://www.unt.edu/vwbk/admission.htm</a><a href="http://www.unt.edu/appupdate.htm" class="DefaultLink"></a>.&nbsp;  Transient students are required to provide an official college transcript from  your most recently completed semester.</p>
	<p>If you have  any questions, you may call Office of Admissions' toll-free hotline at  1-800-UNT-8211.&nbsp; An operator is available M-F, 10 a.m.-7 p.m.  CST.&nbsp;&nbsp;</p>

	<p><strong><a name="FindCurrentOnlineCourses" class="DefaultLink">How can I find current online courses?</a></strong></p>
	<p>There are  two ways to check which courses are available online for the current or  upcoming semester.</p>
	<p>Once the  Schedule of Classes has been released, you can check online course listings on <a href="http://my.unt.edu/" class="DefaultLink">http://my.unt.edu</a>.&nbsp;For  help, see <a href="http://registrar.unt.edu/how-to-view-web-based-classes-online-with-myunt">http://registrar.unt.edu/how-to-view-web-based-classes-online-with-myunt</a>.</p>
	<p>You can also  go to the main <a href="http://www.untecampus.com/" class="DefaultLink">www.UNTeCampus.com</a> and click on &quot;Courses.&quot;&nbsp; Select  the correct semester, then search for courses within the specific college and  department.&nbsp;UNTeCampus is a good resource for online course information,  such as required meeting dates/locations (if any) and textbook  information.<strong> Not all online courses or sections have been listed by faculty so the best place to view current courses (online or on campus) is to view the Schedule of Classes </strong>-- either through <a href="http://my.unt.edu">http://my.unt.edu</a> or by using the Registrar's website to view the schedule without signing in at <a href="http://registrar.unt.edu/registration/schedule-of-classes">http://registrar.unt.edu/registration/schedule-of-classes</a> (you will <strong>only</strong> be able to register for these courses/sections using <a href="http://my.unt.edu">http://my.unt.edu</a>).</p>
	<p><a href="#Top" class="DefaultLink">&lt;&lt; Back to top</a></p>
	<p>&nbsp;</p>
	<p><strong><a name="FindFutureOnlineCourses" class="DefaultLink">How can I find which online courses will be offered in future semesters?</a></strong></p>
	<p>On the <a href="http://www.untecampus.com/" class="DefaultLink">UNTeCampus</a> Web site, you can search for a specific course under &quot;Search&quot; on the  left-hand menu or you can browse offerings per semester by department under  &quot;Courses.&quot;&nbsp; Our goal is to have the information for the upcoming  semester complete by the time early registration occurs (in April for  Summer/Fall and October for Spring).&nbsp; In the meantime, you should contact  the appropriate department for inquiries about specific courses that do not  currently appear on UNTeCampus.&nbsp;</p>
	<p>You can also  view the <a href="http://registrar.unt.edu/registration/schedule-of-classes" class="DefaultLink">Schedule of Classes</a> for information on future courses. Instructions:&nbsp;<a href="http://registrar.unt.edu/how-to-view-web-based-classes-online-with-myunt">http://registrar.unt.edu/how-to-view-web-based-classes-online-with-myunt</a>.</p>

	<p><strong><a name="MyUNTnotinEcampus" class="DefaultLink">Why do I see a course listed in the Schedule of Classes (or MyUNT) that does not appear in UNTeCampus?</a></strong></p>
	<p>In order to provide  you with highly-detailed information, <a href="http://www.untecampus.com" class="DefaultLink">UNTeCampus</a> is updated by the  faculty each semester in time for early registration.&nbsp; The information in  the online <a href="http://registrar.unt.edu/registration/schedule-of-classes" class="DefaultLink">Schedule of Classes</a> comes from  the data that the registrar receives from the department.&nbsp; Sometimes there  are discrepancies when a course is canceled but not removed from the UNTeCampus  listing or when the course has not yet been submitted to UNTeCampus by the  instructor.&nbsp; In the case of such discrepancies, you may contact the  instructor or the specific department for course details, or you may <a href="mailto:untecampus@unt.edu" class="DefaultLink">contact  us</a>.&nbsp; </p>
	<p>UNT  Personnel Directory: <a href="http://info.unt.edu" class="DefaultLink">http://info.unt.edu</a> </p>
	<p>UNT  Departmental Directory: <a href="http://www.unt.edu/find-people-department.htm" class="DefaultLink">http://www.unt.edu/find-people-department.htm</a> </p>
	<p>In general,  the most current information on <strong>course availability</strong> can be obtained from  the online Schedule of Classes at MyUNT &lt;<a href="http://my.unt.edu" class="DefaultLink">http://my.unt.edu</a>&gt;, while  detailed course information, such as required meeting dates/locations and  textbook information, can be obtained at UNTeCampus.</p>

	<p><strong><a name="FindTuitionFee" class="DefaultLink">How do I find tuition and/or fee information for online courses?</a></strong></p>
	<p>If you live  within the state of Texas,  online courses generally cost the same amount as on-campus courses; the regular  resident or non-resident tuition applies.&nbsp;&nbsp; </p>
	<p>Tuition and  Fee Rates:&nbsp; &lt;<a href="http://essc.unt.edu/saucs/tuition.htm" class="DefaultLink">http://essc.unt.edu/saucs/tuition.htm</a>&gt;</p>
	<p>However, if  you currently reside out of state and do not have Texas residency, then a teaching fee is  assigned in lieu of tuition.&nbsp; Because our state funding does not reimburse  the University for providing instruction to students residing outside of Texas, the individual  departments set a teaching fee that must cover the cost of instruction.&nbsp;  The teaching fee is generally less than out-of-state tuition.&nbsp; For  specific course tuition/fee information, you should contact the specific  department offering that course.&nbsp;&nbsp;</p>
	<p>Departmental  Directory&nbsp; &lt;<a href="http://www.unt.edu/departmentscontactinfo.htm" class="DefaultLink">http://www.unt.edu/departmentscontactinfo.htm</a>&gt;</p>
	<p>In addition,  if you are taking only online courses during the semester, several fees such as  the student recreation center fee and student union fee are waived.&nbsp;</p>

	<p>You should  see your online course listed on the left.&nbsp; Click on them to enter each  one.&nbsp; You will see things such as a calendar, assignments, lectures, and a  syllabus.&nbsp; Read through the syllabus and any introductory comments the  instructor may have left.&nbsp; You will find information on expectations  regarding the frequency of reading, accessing the Web site, posting comments,  and so forth.&nbsp; Also, check the calendar or schedule for the next week for  dates of assignments and any face-to-face meetings, if your course has  any.&nbsp;&nbsp;</p>

	<p><strong><a name="F2FMeetings" class="DefaultLink">How do I know when and/or where a course meets? If some site-based meetings are required, how do I find out where to show up?</a></strong></p>
	<p>First, look  for the information on the <a href="http://www.untecampus.com/" class="DefaultLink">UNTeCampus</a> Web site.&nbsp;  Search for the course under &quot;Search.&quot;&nbsp; Schedules and  face-to-face meeting dates, times, and locations may appear here.&nbsp; If  nothing appears in this area or it is listed as TBA, you should confirm meeting  times and locations with your instructor.&nbsp;</p>
	<p>Then, log  into Blackboard Learn and search for this information (usually available in the  calendar or schedule).&nbsp; You can also email your instructor within Blackboard Learn if  this course location information does not appear.</p>
	<p>If you  cannot find information using either of the above methods, go to the course  details section on <a href="http://my.unt.edu/" class="DefaultLink">http://my.unt.edu</a>.</p>
	<p>Log into the  Web site using your <a href="https://ams.unt.edu/" class="DefaultLink">EUID</a>.&nbsp; Then select &quot;Academics&quot; from the  left-hand menu.&nbsp; Then select &quot;Classes and Schedules,&quot; then  &quot;Schedule of Classes.&quot;&nbsp; Input the institution, career, and term  information and you will be taken to the Course Search page.&nbsp;</p>
	<p>On this  page, enter the course prefix and the catalog number.&nbsp; Once the course  comes up, find the specific section for which you registered.&nbsp; Under the  course name and number there will be a section that lists any meeting date,  time, and locations for the course.&nbsp; If a time and date are listed under  the course details, but no meeting place, contact your instructor or the  department for details (see below).&nbsp; If you are not familiar with the  campus, you can find building locations and abbreviations listed on the <a href="http://www.unt.edu/pais/map/campusmap.htm" class="DefaultLink">Campus Map</a>.&nbsp;</p>
	<p>If  &quot;TBA&quot; appears here in the course details instead, it is likely that  there are no face-to-face meetings. &nbsp;However, you should contact your  instructor or the department to confirm (see links below).&nbsp;</p>
	<p>UNT  Personnel Directory: <a href="http://info.unt.edu/" class="DefaultLink">http://info.unt.edu</a><strong> </strong></p>
	<p>Department  Contact Information: <a href="http://www.unt.edu/departmentscontactinfo.htm" class="DefaultLink">http://www.unt.edu/departmentscontactinfo.htm</a></p>

	<p><strong><a name="FindTextbooks" class="DefaultLink">How do I find and order textbooks?</a></strong></p>
	<p>You can check  two locations for this information; occasionally there is conflicting  information, and if that's the case, you'll need to contact the instructor  about the correct text for the course.&nbsp;</p>
	<p>Go to the <a href="http://maps.unt.edu/departments/3448/view" class="DefaultLink">UNT  Bookstore</a> &lt;<a href="http://unt.bncollege.com/webapp/wcs/stores/servlet/BNCBHomePage?storeId=71237&amp;catalogId=10001&amp;langId=-1" class="DefaultLink">http://unt.bncollege.com/webapp/wcs/stores/servlet/BNCBHomePage?storeId=71237&amp;catalogId=10001&amp;langId=-1</a>&gt;,  and enter the appropriate information for your course.&nbsp; Textbooks should  be listed here; you can order them from the bookstore site or you can get the  information and purchase them elsewhere.</p>
	<p>On the <a href="http://www.untecampus.com/" class="DefaultLink">UNTeCampus</a> Web site, find the course using the &quot;Search&quot; function.&nbsp; Click on  the course title and look near the bottom for textbook information.</p>
	<p>Some  instructors may provide information about textbooks on Blackboard Learn, but choose  not to leave this information with the UNT Bookstore or UNTeCampus, so be sure  to check.&nbsp; If no texts are listed at the UNT Bookstore or UNTeCampus, be  sure to confirm this information with your instructor before or during the  first week of classes.&nbsp;</p>
