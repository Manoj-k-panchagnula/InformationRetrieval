<h2><a name="veterans"></a>Benefits for Veterans</h2>
<table width="100%" cellspacing="0" cellpadding="5" border="0" summary="Benefits for Veterans">
    <tbody>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="width:68%">
            <h3><a href="http://www.tvc.texas.gov/Service-Connected-Compensation.aspx">Service-connected Disability Compensation</a></h3>
            </td>
            <td scope="col" style="width:32%">provided by:<br />
            <a href="http://www.tvc.texas.gov/">Texas Veterans Commission</a></td>
        </tr>
        <tr>
            <td valign="top" colspan="2">Service-connected Disability Compensation is a monetary benefit paid to veterans who are disabled by an injury or illness that was incurred or aggravated during active military service. Disability compensation varies with the degree of disability and the number of veteran’s dependents, and is paid monthly. Veterans with certain severe disabilities may be eligible for additional special monthly compensation.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://www.tvc.texas.gov/Non-Service-Connected-Pension.aspx">Non Service-connected Pension</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://www.tvc.texas.gov/">Texas Veterans Commission</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">Non Service-connected Pension is a monthly benefit paid to wartime veterans, 65 years or older, who have limited or no income. Wartime veterans, younger than 65, who are permanently and totally disabled may also be eligible.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col">
            <h3><a href="http://www.tvc.texas.gov/Appeals.aspx">VA Appeals Assistance</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://www.tvc.texas.gov/">Texas Veterans Commission</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">You have the right to appeal VA decisions. The Texas Veterans Commission can help veterans with the appeals process.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://www.tvc.texas.gov/Claims.aspx">Claims Representation and Counseling</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://www.tvc.texas.gov/">Texas Veterans Commission</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">The Texas Veterans Commission is the designated agency of the State of Texas to represent the State and its veterans before the U.S. Department of Veterans Affairs (VA). The agency represents veterans in filing VA disability claims and during VA appeals processes, and assists dependents with survivor benefits.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://gibill.va.gov/">GI Bill</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://va.gov/">U.S. Dept. of Veterans Affairs</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">The GI Bill is a federal education benefit provided by the Department of Veterans Affairs available for eligible veterans and service members.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://www.collegefortexans.com/apps/financialaid/tofa2.cfm?ID=500">Hazlewood Exemption</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://www.tvc.texas.gov/">
            Texas Veterans Commission</a><a href="http://www.thecb.state.tx.us/"><br />
            </a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">The Hazlewood Exemption (Hazlewood Act) provides education benefits to honorably discharged or separated Texas veterans and to eligible dependent children and spouses of Texas veterans.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://www.vba.va.gov/bln/vre/">Vocational Rehabilitation &amp; Employment Program</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://va.gov/">U.S. Dept. of Veterans Affairs</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">The Vocational Rehabilitation and Employment program provides educational and vocational counseling to service members, veterans, and certain dependents at no charge. These counseling services are designed to help an individual choose a vocational direction, determine the course needed to achieve the chosen goal, and evaluate the career possibilities open to them
            <p> </p>
            </td>
        </tr>
    </tbody>
</table>
<h2><a name="spouses"></a>Benefits for Spouses &amp; Dependents</h2>
<table width="100%" cellspacing="0" cellpadding="5" border="0" summary="Benefits for Spouses/Dependents">
    <tbody>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="width:68%">
            <h3><a href="http://www.vba.va.gov/bln/dependents/spouse.htm">Surviving Spouse Benefits - Dependency and Indemnity Compensation (DIC)</a></h3>
            </td>
            <td scope="col" style="width:32%">provided by:<br />
            <a href="http://va.gov/">U.S. Dept. of Veterans Affairs</a></td>
        </tr>
        <tr>
            <td valign="top" colspan="2">VA may provide a monthly benefit to the surviving spouse and some parents of a service member or veteran whose death was service-related.
            <p> </p>
            </td>
        </tr>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="background-color:#e6e6e6">
            <h3><a href="http://www.vba.va.gov/bln/dependents/spouse.htm">Surviving Spouse Pension</a></h3>
            </td>
            <td scope="col">provided by:<br />
            <a href="http://va.gov/">U.S. Dept. of Veterans Affairs</a></td>
        </tr>
        <tr>
            <td scope="col" colspan="2">VA provides pensions to low-income surviving spouses and unmarried children of deceased veterans with wartime service. The pension provides a monthly payment to bring an eligible person’s income to a level established by law.
            <p> </p>
            </td>
        </tr>
    </tbody>
</table>
<h2><a name="burial"></a>Burial Benefits </h2>
<table width="100%" cellspacing="0" cellpadding="5" border="0" summary="Burial Benefits">
    <tbody>
        <tr bgcolor="#e6e6e6">
            <td scope="col" style="width:68%">
            <h3><a href="http://www.cem.va.gov/bbene_burial.asp">VA Headstones and Markers</a></h3>
            </td>
            <td scope="col" style="width:32%">provided by:<br />
            <a href="http://www.cem.va.gov/">Dept. of Veterans Affairs</a></td>
        </tr>

			function detectLang() {
				// local = if path starts with /en or /es; translated = request param has a tl= in it
				var l, p = window.location.pathname, q = window.location.search, local = new RegExp('^/(en|es)\/?','i'), translated = new RegExp('(&|/\?)tl=([a-z]{2})','gi');
				switch(true) {
					// if path starts with /en or /es set var to en or es
					case local.test(p):
						l = p.substring(1,3);
						break;
					// if request params have tl= in it set var to value of tl param
					case translated.test(q):
						var i = q.indexOf('tl=');
						l = q.substring(i+3, i+5);
						break;
					// if nothing else, set lang to en
					default:
						l = 'en';
						break;
				}
				return l;
			}
