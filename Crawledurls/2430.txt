      map: gmap,
    });

      navigator.geolocation.getCurrentPosition(
        function(position) {
          var userLocation = new google.maps.LatLng( position.coords.latitude, position.coords.longitude );
          var userMarker = new google.maps.Marker(
            {
              position: userLocation,
              visible: true,
              map: gmap,
            }
          );
