		// callback to remove scheduled blank in the native ad slot
		function blankAdCallback(width, height, pos)
		{
			var found = false;
			// loop through ads
			for(var i = 0; i < dfpAdSlots.length; i++)
			{
				// find top native
				if(dfpAdSlots[i].css == "native"
					&& dfpAdSlots[i].dfpAd
					&& dfpAdSlots[i].dfpAd.getTargeting("pos") == pos)
				{
					// loop through it's sizes
					for(var j = 0; j < dfpAdSlots[i].sizes.length; j++)
					{
						// confirm sizes match
						if(dfpAdSlots[i].sizes[j][0] == width
							&& dfpAdSlots[i].sizes[j][1] == height)
						{
							found = true;

							// make sure elements on the page
							var element = document.getElementById(dfpAdSlots[i].id);
							if(!element) break;

							// move element to last child so its hidden
							element = element.parentElement;
							element.parentNode.appendChild(element);
						}
					}
				}

				if(found)
				{
					break;
				}
			}
		}
