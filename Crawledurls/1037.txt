        var minutes = 0;
        var seconds = 0;
        function decrement_timers() {

            seconds--;

            if(seconds < 0) {
                seconds = 59;
                minutes--;
            }

            var string = (seconds < 10) ? minutes + ':0' + seconds : minutes + ':' + seconds;

            $('.cart-timer').html(string);

            if(!(minutes === 0 && seconds === 0)) {
                setTimeout("decrement_timers()", 1000);
            }
            else {
                timer_expired();
            }
        }

        function timer_expired() {
            AccountManager.DialogManager.release_all_dialogs();
            $.ajax({ url: client_url + "cart/reset",
                success: function(data){

                                            message = AccountManager.tu('TR_TIMER_EXPIRED_MSG');
                    
                    var options = {
                        init: {
                            title        : "Timer Has Expired",
                            content      : message,
                            closeOnEscape: false,
                            hideClose    : true,
                            height       : 200,
                            buttons      : {
                                confirm: {
                                    text : "start over",
                                    click: { redirect: 'buy/browse' }
                                }
                            }
                        }
                    };
                    AccountManager.DialogManager.message_dialog(options);
                }
            });
        }

        function init_timers() {
            $.ajax({
                url: client_url + "timer/remaining_time",
                type: 'get',
                dataType: 'json',
                success: function(data){
                    var code = data.command1.control.result.code;

                    if (code == 0) {
                        var time = data.command1.data.remaining_time.split(':');
                        minutes = parseInt(time[0]);
                        seconds = parseInt(time[1]);
                    }

                    if(!(minutes === 0 && seconds === 0)) {
                        setTimeout("decrement_timers()", 1000);
                    }
                    else {
                        timer_expired();
                    }
                }
            });
        }

                function handle_general_ajax_response_code(code, message)
{
    var url = null;

    switch (code)
    {
        case 'FATAL_ERROR':
        case 'ENTITY_MANAGER':
        case 'ENTITY_MANAGER_RESOLVER':
        case 'PROXY_LAZY_LOAD_MISSING_INFO':
        case 'BROKEN_ICE':
        case "SECURITY_TOKEN_INVALID":
            url = 'error/104';
            break;
        case 'ACL_DENIED':
        case 'ACCES_DENIED':
            url = 'account/logout/ACCESS_DENIED';
            break;
        case 'SESSION_EXPIRED':
        case 'SESSION_ERROR':
            url = 'account/logout/SESSION_EXPIRED';
            break;
        case 'BACKEND_ERROR':
        case 'BACKEND_DOWN':
        case "PAYMENT_PENDING":
        case "DUPLICATE_PAYMENT":
        case "BACKEND_VALIDATION_ERROR":
        case "SYSTEM_ACCEPT_NO_ONLINE_CC_PAYMENT":
